Unemployment rate by county, August 2016.
d3.csv('unemployment-x.csv', d => +d.rate)
.then(data => {
  var margin = { top: 50, left: 100, bottom: 20, right: 50 },
    width = 600 - margin.left - margin.right,
    height = 400 - margin.top - margin.bottom;

  var svg = d3.select('#chart')
    .attr('width', width + margin.left + margin.right)
    .attr('height', height + margin.top + margin.bottom)
    .append('g')
    .attr('transform', 'translate(' + margin.left + ', ' + margin.top + ')');

  var x = d3.scaleLinear()
    .domain([0, d3.max(data)])
    .range([0, width])
    .nice();

  //see https://observablehq.com/@d3/d3-bin
  var bin = d3.bin();

  //see https://observablehq.com/@d3/scale-ticks
  var thresholds = x.ticks(20);

  //see https://observablehq.com/@d3/d3-bin and look for bin.domain and bin.thresholds
  var bin = d3.bin()
    .domain(x.domain()) 
    .thresholds(thresholds);

  //see https://observablehq.com/@d3/d3-bin and look for bin.domain and bin.thresholds
  var bins = bin(data);

  //see https://observablehq.com/@d3/d3-scalelinear and look for scale.nice
  var y = d3.scaleLinear()
    .domain([0, d3.max(bins, d => d.length)])
    .range([height, 0])
    .nice();

  var bar = svg.selectAll('.bar')
    .data(bins)  //data join with bins
    .enter()
    .append('g')
    .attr('class', 'bar')
    .attr('transform', d => 'translate(' + x(d.x0) + ',' + y(d.length) + ')');

  bar.append('rect')
    .attr('x', 0)
    .attr('width', d => x(d.x1) - x(d.x0))
    .attr('height', d => height - y(d.length));

  var xAxis = d3.axisBottom()
    .scale(x);

  svg.append('g')
    .attr('class', 'axis')
    .attr('transform', 'translate(0,' + height + ')')
    .call(xAxis);

  var yAxis = d3.axisLeft()
    .scale(y)
    .ticks(10, '.0f');

  svg.append('g')
    .attr('class', 'axis')
    .call(yAxis);

  svg.append('text')
    .attr('x', 5)
    .attr('y', 5)
    .attr('class', 'label')
    .text('Counties');

  svg.append('text')
    .attr('x', 0.75 * width)
    .attr('y', height - 5)
    .attr('class', 'label')
    .text('Unemployment (%)');
});
